99       private class LoginTask extends AsyncTask<String, Void, String> { 
100   
101          @Override 
102          protected String doInBackground(String... strings) { 
103   
104              //Clique no botao de Registo 
105              String message, Userfeedback; 
106              // Pedido Ã  base de dados 
107              ClientJava clientRegister = new ClientJava(ServerIP, new IPandPORT().PHPServer_Port); 
108              clientRegister.send_message("ANDROID LOGIN " + strings[0] + " " + strings[1] + "\n"); 
109              message = clientRegister.receive_message(); 
110              try { 
111                  clientRegister.Close(); 
112              } catch (IOException e) { 
113                  e.printStackTrace(); 
114              } 
115              //message = "NOK PASSWD"; 
116              //Se a comunicacao c/base de dados for bem sucedida 
117              if(message!=null) { 
118                  if (message.regionMatches(0, "OK", 0, 2)) { 
119                      String[] parts = message.split(" "); 
120                      PANid = parts[1]; 
121                      return Userfeedback = "Login Successful!"; 
122   
123                  } else if (message.regionMatches(0, "NOK EMAIL", 0, 9)) { 
124                      return Userfeedback = "Email address not found."; 
125                  } else if (message.regionMatches(0, "NOK PASSWD", 0, 10)) { 
126                      return Userfeedback = "Password does not match Email."; 
127                  } 
128              } 
129              return ""; 
130          } 
131          protected void onPostExecute(String UserFeedback) { 
132   
133              Toast.makeText(getApplicationContext(), UserFeedback, Toast.LENGTH_SHORT).show(); 
134              if (UserFeedback.equals("Login Successful!")){ 
135                  Intent UserAreaIntent = new Intent(MainActivity.this, UserArea.class); 
136                  UserAreaIntent.putExtra("ServerIP",ServerIP); 
137                  UserAreaIntent.putExtra("Email",Email); 
138                  UserAreaIntent.putExtra("PANid",PANid); 
139   
140                  MainActivity.this.startActivity(UserAreaIntent); 
141              } 
142   
143          }
