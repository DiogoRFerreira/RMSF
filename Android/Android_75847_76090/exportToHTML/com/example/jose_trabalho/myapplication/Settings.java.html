156      private class ModifyTask extends AsyncTask<String, Void, String> { 
157   
158          @Override 
159          protected String doInBackground(String... strings) { 
160   
161              //Clique no botao de Registo 
162              String message; 
163              // Pedido à base de dados 
164              ClientJava clientRegister = new ClientJava(ServerIP, new IPandPORT().PHPServer_Port); 
165              switch (strings[3]) { 
166                  case "GenSettings": 
167                      clientRegister.send_message("ANDROID MODIFY " + strings[0] + " " + strings[1] + " " + (Boolean.parseBoolean(strings[2]) ? 1 : 0) + "\n"); 
168                      break; 
169                  case "Sensor": 
170                      clientRegister.send_message("ANDROID MODIFY " + strings[0] + " SENSOR " + strings[1] +" " + strings[2] + "\n"); 
171                      break; 
172              } 
173              message = clientRegister.receive_message(); 
174              Log.d("Message rcv: ", message); 
175              try { 
176                  clientRegister.Close(); 
177              } catch (IOException e) { 
178                  e.printStackTrace(); 
179              } 
180              //message = "NOK PASSWD"; 
181              //Se a comunicacao c/base de dados for bem sucedida 
182              if (message != null) { 
183                  if (message.regionMatches(0, "OK", 0, 2)) { 
184                      return strings[0] + " modified"; 
185                  } else return "Error ocurred"; 
186              } 
187              return "Failed communication with server"; 
188          }

196      private class GetSettingsTask extends AsyncTask<String, Void, String> { 
197   
198          @Override 
199          protected String doInBackground(String... strings) { 
200              String message; 
201              // Pedido à base de dados 
202              ClientJava clientRegister = new ClientJava(ServerIP, new IPandPORT().PHPServer_Port); 
203              // strings[0] = PAN ID 
204              clientRegister.send_message("ANDROID RETRIEVE " + strings[0] + "\n"); 
205              message = clientRegister.receive_message(); 
206              try { 
207                  clientRegister.Close(); 
208              } catch (IOException e) { 
209                  e.printStackTrace(); 
210              } 
211              //Se a comunicacao c/base de dados for bem sucedida 
212              if (message != null) { 
213                  if (message.regionMatches(0, "OK", 0, 2)) { 
214                      String[] parts = message.split(" "); 
215                      AlarmSysEnabled = Integer.parseInt(parts[1]) != 0; 
216                      Buzzer = Integer.parseInt(parts[2]) != 0; 
217                      Propagation = Integer.parseInt(parts[3]) != 0; 
218                      int j = 0; 
219                      for (int i = 4; i < parts.length-1; i+=2) { 
220                          sensors[j].ID = parts[i]; 
221                          sensors[j].enabled = Integer.parseInt(parts[i + 1]) != 0; 
222                          j++; 
223                      } 
224                  } 
225              } 
226              return ""; 
227   
228          } 
229          protected void onPostExecute(String UserFeedback) { 
230   
231          } 
232      }
