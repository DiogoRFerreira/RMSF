6    import java.util.HashMap; 
7    import java.util.List; 
8     
9    import android.content.Context; 
10   import android.graphics.Typeface; 
11   import android.view.LayoutInflater; 
12   import android.view.View; 
13   import android.view.ViewGroup; 
14   import android.widget.BaseExpandableListAdapter; 
15   import android.widget.TextView; 
16    
17   public class ExpandableListAdapter extends BaseExpandableListAdapter { 
18    
19       private Context _context; 
20       private List<String> _listDataHeader; // header titles 
21       // child data in format of header title, child title 
22       private HashMap<String, List<String>> _listDataChild; 
23    
24       public ExpandableListAdapter(Context context, List<String> listDataHeader, 
25                                    HashMap<String, List<String>> listChildData) { 
26           this._context = context; 
27           this._listDataHeader = listDataHeader; 
28           this._listDataChild = listChildData; 
29       } 
30    
31       @Override 
32       public Object getChild(int groupPosition, int childPosititon) { 
33           return this._listDataChild.get(this._listDataHeader.get(groupPosition)) 
34                   .get(childPosititon); 
35       } 
36    
37       @Override 
38       public long getChildId(int groupPosition, int childPosition) { 
39           return childPosition; 
40       } 
41    
42       @Override 
43       public View getChildView(int groupPosition, final int childPosition, 
44                                boolean isLastChild, View convertView, ViewGroup parent) { 
45    
46           final String childText = (String) getChild(groupPosition, childPosition); 
47    
48           if (convertView == null) { 
49               LayoutInflater infalInflater = (LayoutInflater) this._context 
50                       .getSystemService(Context.LAYOUT_INFLATER_SERVICE); 
51               convertView = infalInflater.inflate(R.layout.list_item, null); 
52           } 
53    
54           TextView txtListChild = (TextView) convertView 
55                   .findViewById(R.id.lblListItem); 
56    
57           txtListChild.setText(childText); 
58           return convertView; 
59       } 
60    
61       @Override 
62       public int getChildrenCount(int groupPosition) { 
63           return this._listDataChild.get(this._listDataHeader.get(groupPosition)) 
64                   .size(); 
65       } 
66    
67       @Override 
68       public Object getGroup(int groupPosition) { 
69           return this._listDataHeader.get(groupPosition); 
70       } 
71    
72       @Override 
73       public int getGroupCount() { 
74           return this._listDataHeader.size(); 
75       } 
76    
77       @Override 
78       public long getGroupId(int groupPosition) { 
79           return groupPosition; 
80       } 
81    
82       @Override 
83       public View getGroupView(int groupPosition, boolean isExpanded, 
84                                View convertView, ViewGroup parent) { 
85           String headerTitle = (String) getGroup(groupPosition); 
86           if (convertView == null) { 
87               LayoutInflater infalInflater = (LayoutInflater) this._context 
88                       .getSystemService(Context.LAYOUT_INFLATER_SERVICE); 
89               convertView = infalInflater.inflate(R.layout.list_group, null); 
90           } 
91    
92           TextView lblListHeader = (TextView) convertView 
93                   .findViewById(R.id.lblListHeader); 
94           lblListHeader.setTypeface(null, Typeface.BOLD); 
95           lblListHeader.setText(headerTitle); 
96    
97           return convertView; 
98       } 
99    
100      @Override 
101      public boolean hasStableIds() { 
102          return false; 
103      } 
104   
105      @Override 
106      public boolean isChildSelectable(int groupPosition, int childPosition) { 
107          return true; 
108      } 
109  }
