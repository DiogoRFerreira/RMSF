26    
27               //Clique no botao de Registo 
28               String message, Userfeedback; 
29               // Pedido Ã  base de dados 
30               ClientJava clientRegister = new ClientJava(ServerIP, new IPandPORT().PHPServer_Port); 
31               clientRegister.send_message("ANDROID REGISTER " + strings[0] + " " + strings[1] + " " + strings[2] + " " + strings[3]+"\n"); 
32               message = clientRegister.receive_message(); 
33               try { 
34                   clientRegister.Close(); 
35               } catch (IOException e) { 
36                   e.printStackTrace(); 
37               } 
38               //message = "NOK PASSWD"; 
39               //Se a comunicacao c/base de dados for bem sucedida 
40               if(message!=null) { 
41                   if (message.regionMatches(0, "OK", 0, 2)) { 
42                       return Userfeedback = "Successfully registered!"; 
43                   } else if (message.regionMatches(0, "NOK PANID", 0, 9)) { 
44                       return Userfeedback = "PAN Identifier does not exist."; 
45                   } else if (message.regionMatches(0, "NOK PANSK", 0, 9)) { 
46                       return Userfeedback = "Serial Key does not match the inserted PAN."; 
47                   } 
48               } 
49               return ""; 
50           }
